\apisummary{
    A synchronization call for a particular task
}

\begin{apidefinition}

\begin{Csynopsis}
    ResultsBuffer* sync(Handle qh)
\end{Csynopsis}

\begin{Cppsynopsis}
    ResultsBuffer* sync(Handle qh)
\end{Cppsynopsis}


\begin{apiarguments}
    \apiargument{IN}{qh}{Handle for the task launched by the taskInitiate call}
\end{apiarguments}

\apidescription{
    This a blocking execution call.  The sync call is initiated from the host device.  It blocks until the terminating condition of the objective function has been met.
}

\apireturnvalues{
    Returns an instance of \DATATYPENAME{ResultsBuffer}.
}

}

\end{apidefinition}
