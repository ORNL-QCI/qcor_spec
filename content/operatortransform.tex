\apisummary{
    Performs general transformations on Operators. 
}

\begin{apidefinition}

\begin{Csynopsis}
Operator* operatorTransform(char* modifier, Operator* Op)
\end{Csynopsis}

\begin{Cppsynopsis}
Operator* operatorTransform(std::string modifier, Operator& Op)
\end{Cppsynopsis}


\begin{apiarguments}
    \apiargument{IN}{modifier}{A string indicating the modifier name}
    \apiargument{IN}{Op}{an Operator }
\end{apiarguments}

\apidescription{
    \FUNC{operatorTransform} performs a transformation mapping \DATATYPENAME{Operator}, \VAR{Op}, to a new \DATATYPENAME{Operator} as denoted by the string \VAR{modifier} which indicates the transformation method. 
}

\apireturnvalues{
    Returns a pointer to the new transformed \DATATYPENAME{Operator}.
}      

\apinotes{

}

\end{apidefinition}
