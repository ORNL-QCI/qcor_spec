\apisummary{
    Abstraction which models general transformations on operators. 
}

\begin{apidefinition}

\begin{Csynopsis}
Operator* operatorTransform(char* modifier, Operator* Op)
\end{Csynopsis}

\begin{Cppsynopsis}
Operator* operatorTransform(std::string modifier, Operator& Op)
\end{Cppsynopsis}


\begin{apiarguments}
    \apiargument{IN}{modifier}{A string indicating the modifier name}
    \apiargument{IN}{Op}{an Operator type}
\end{apiarguments}

\apidescription{
    A transformation mapping operator, \VAR{Op}, to a new operator as denoted by the string \VAR{modifier} which indicates the transformation method. 
}

\apireturnvalues{
    pointer to the resulting operator 
}      

\apinotes{

}

\end{apidefinition}
