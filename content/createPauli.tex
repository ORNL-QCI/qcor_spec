\apisummary{
    Creates a Pauli observable.
}

\begin{apidefinition}

\begin{Csynopsis}
    Observable createPauli(SiteMap sitemap, double complex coef, char* paulistr)
\end{Csynopsis}

\begin{Cppsynopsis}
    Observable qcor::createPauli(std::map<int,string> sitemap, std::complex coef, string paulistr)
\end{Cppsynopsis}

\begin{apiarguments}
    \apiargument{IN}{sitemap}{mapping of an integer to a string of value``x'', ``y'' or ``z''}
    \apiargument{IN}{coef}{coefficient for sitemap terms}
    \apiargument{IN}{paulistr}{string representation of Pauli operator}
\end{apiarguments}

\apidescription{
        \FUNC{createPauli} accepts a string representation of a complex Pauli operator, \VAR{paulistr}, or an integer:string mapping, \VAR{sitemap}, of an integer site number to a string ``x'', ``y'', or ``z''. The complex coefficient \VAR{coef} argument can optionally be included with \VAR{sitemap} to multiply each term in the site map. If \VAR{coef} is not included, the default value is 1. If \VAR{paulistr} is provided, the expected form of the string is ``<coefficient> <pauli string> <integer> ... <optional operation>''. If both \VAR{paulistr} and \VAR{sitemap} are provided, \VAR{paulistr} is ignored and \VAR{sitemap} is used to produce the Pauli observable. If neither \VAR{paulistr} or \VAR{sitemap} are provided, the identy map is used to create the Pauli observable.
