\apisummary{
    Creates a Pauli X operator at site n.   
}

\begin{apidefinition}

\begin{Csynopsis}
    PauliOperator pauliX(int n)
\end{Csynopsis}

\begin{Cppsynopsis}
    PauliOperator qcor::pauliX(int n)
\end{Cppsynopsis}


\begin{apiarguments}
    \apiargument{IN}{n}{integer value}
\end{apiarguments}

\apidescription{
       \FUNC{pauliX} accepts an integer input that creates a Pauli X operator at site \VAR{n}.  The Pauli X operator is returned as a \DATATYPENAME{PauliOperator}.
}

\apireturnvalues{
    Returns a single \DATATYPENAME{PauliOperator}.
}
